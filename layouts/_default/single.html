{{ define "main" }} {{/* single.html — Single content page layout This template renders individual content pages. Blog-specific features (metadata, table of contents, post navigation) only appear on pages that are NOT structural content (projects, footer badges). Structural content is identified by its category membership in taxonomy_exclude. Key guards: $is_blog_post — true for non-structural pages with a publication date $is_structural — true for pages in excluded categories (projects, footer) .Params.toc — opt-in table of contents for Markdown posts */}} {{/* Determine if this page is structural content (e.g., project profiles, footer badges). These pages are used for layout composition, not as standalone blog posts. We identify them two ways: 1. Their categories overlap with taxonomy_exclude (via intersect) 2. They set hide_sitemap: true (a convention for embedded content that shouldn't appear as independent pages in search results) */}} {{- $exclude := site.Params.taxonomy_exclude | default slice -}} {{- $is_structural_by_category := and .Params.categories (gt (len (intersect $exclude .Params.categories)) 0) -}} {{- $is_structural := or (.Params.hide_sitemap | default false) $is_structural_by_category -}} {{- $is_blog_post := and (not .Date.IsZero) (not $is_structural) -}} {{/* --- Cover image (between title and metadata) --- */}} {{/* Renders the post's cover image from .Params.images (Hugo's standard convention). Using .Params.images also enables the built-in OpenGraph and Twitter Card templates to use the image for social sharing. Supports three path formats: - Remote URL (e.g., "https://cdn.example.com/photo.jpg") — used as-is - Absolute path (e.g., "/img/photo.jpg") — served from static/img/ - Filename only (e.g., "photo.jpg") — resolved relative to the post's content directory */}} {{ if and $is_blog_post .Params.images }} {{- $src := index .Params.images 0 -}} {{- $alt := .Title | plainify -}} {{- if and (not (hasPrefix $src "/")) (not (findRE "^https?://" $src)) -}} {{/* Non-absolute path — resolve relative to the post's directory. Blog images are stored alongside the post files in content/blog/YYYY/MM/. Since posts are standalone files (not leaf bundles), .Resources is not available. We construct the URL from the post's directory path. */}} {{- with .File -}} {{- $src = path.Join "/" (path.Dir .Path) $src -}} {{- end -}} {{- end -}}
{{ end }} {{/* --- Blog post metadata (date, author, reading time, tags) --- */}} {{ if $is_blog_post }} {{- partial "post-meta.html" . -}} {{ end }} {{/* --- Table of contents (Markdown only, opt-in via front matter) --- */}} {{/* Toph provides unified ToC styling for both Markdown and AsciiDoc. AsciiDoc: Uses its native :toc: document attribute. When the site config sets markup.asciidocExt.preserveTOC: true, Asciidoctor renders the ToC directly inside .Content using div#toc > div#toctitle + ul.sectlevel1 markup. No template action is needed — the theme CSS styles it. Markdown: Opt-in via "toc: true" in front matter. Hugo's .TableOfContents is wrapped in #toc/#toctitle markup to match the AsciiDoc structure, so both formats share the same CSS. We detect AsciiDoc files to avoid rendering a duplicate ToC alongside the one already in .Content. */}} {{- $isAsciidoc := and .File (eq .File.Ext "adoc") -}} {{ if and .Params.toc (not $isAsciidoc) }} {{ with .TableOfContents }}
Table of Contents
{{ . }}
{{ end }} {{ end }} {{/* Heading anchors — clickable 🔗 links next to section headings. Markdown: Handled by the render-heading.html hook, which injects anchors at render time. No processing needed here. AsciiDoc: Asciidoctor generates headings directly in .Content, bypassing Hugo's render hooks. We use replaceRE to inject the same anchor markup into Asciidoctor-generated headings so both formats get identical anchor behavior. */}} {{ if $isAsciidoc }} {{- .Content | replaceRE `(]*>.+?)()` `${1} 🔗${3}` | safeHTML }} {{ else }} {{- .Content }} {{ end }} {{/* --- Previous/next post navigation --- */}} {{ if $is_blog_post }} {{- partial "post-nav.html" . -}} {{ end }} {{ end }}
